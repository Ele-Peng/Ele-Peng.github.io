    • such as:
      • 0, 1, 1, 2, 3, 5, 8, 13, 21 …

初级实现：简单 for 循化实现

中级实现：简单 for 循化实现

// 递归优化，跳过计算过的中间变量 Time: O(n) Space: O(n)
// {
  let res = [0]
  function fibonacci(n) {
    if (n <= 1) {
      return res[n] = Math.max(n, 0);
    }
    if (res[n]) {
      return res[n];
    }
    res[n] = fibonacci(n - 1) + fibonacci(n - 2);
    return res[n]
  }
  // 求数组第十项的值
  fibonacci(9)
// }
